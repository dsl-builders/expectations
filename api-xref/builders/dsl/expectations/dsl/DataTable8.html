18  package builders.dsl.expectations.dsl;
19  
20  import builders.dsl.expectations.Expectations;
21  import org.junit.jupiter.api.DynamicTest;
22  import org.opentest4j.AssertionFailedError;
23  
24  import java.util.ArrayList;
25  import java.util.List;
26  import java.util.stream.Stream;
27  
28  /**
29   * Represents the data table of the parametrized test with eight columns.
30   * @param <A> the type of the first column
31   * @param <B> the type of the second column
32   * @param <C> the type of the third column
33   * @param <D> the type of the fourth column
34   * @param <E> the type of the fifth column
35   * @param <F> the type of the sixth column
36   * @param <G> the type of the seventh column
37   * @param <H> the type of the eighth column
38   */
39  public class DataTable8<A, B, C, D, E, F, G, H> {
40  
41      private final List<Row8<A, B, C, D, E, F, G, H>> data = new ArrayList<>();
42      private final Headers8 headers;
43  
44      /**
45       * Creates a new data table with eight columns.
46       * @param headers the headers of the data table
47       * @param rows the rows of the data table
48       */
49      public DataTable8(Headers8 headers, Iterable<Row8<A, B, C, D, E, F, G, H>> rows) {
50          this.headers = headers;
51          rows.forEach(data::add);
52      }
53  
54      /**
55       * Creates a new expectation for the data table.
56       * @param template the template of the expectation where the placeholders are the headers of the data table prefixed with <code>#</code>
57       * @param verification the assertion that must return <code>true</code> for the expectation to pass
58       * @return the new expectation that can be used with {@link org.junit.jupiter.api.TestFactory} annotation to generate dynamic tests
59       */
60      public Expectations expect(String template, Assertion8<A, B, C, D, E, F, G, H> verification) {
61          return new Expectations8<>(this, template, verification);
62      }
63  
64      /**
65       * Creates a new expectation for the data table.
66       * @param template the template of the expectation where the placeholders are the headers of the data table prefixed with <code>#</code>
67       * @param verification the assertion that must return <code>true</code> for the expectation to pass
68       * @return the new expectation that can be used with {@link org.junit.jupiter.api.TestFactory} annotation to generate dynamic tests
69       */
70      public Expectations verify(String template, Verification8<A, B, C, D, E, F, G, H> verification) {
71          return new Expectations8<>(this, template, (a, b, c, d, e, f, g, h) -> {
72              verification.verify(a, b, c, d, e, f, g, h);
73              return true;
74          });
75      }
76  
77      /**
78       * Adds a new row to the data table.
79       * @param a the value of the first column
80       * @param b the value of the second column
81       * @param c the value of the third column
82       * @param d the value of the fourth column
83       * @param e the value of the fifth column
84       * @param f the value of the sixth column
85       * @param g the value of the seventh column
86       * @param h the value of the eighth column
87       * @return the data table with the new row
88       */
89      public DataTable8<A, B, C, D, E, F, G, H> and(A a, B b, C c, D d, E e, F f, G g, H h) {
90          data.add(new Row8<>(a, b, c, d, e, f, g, h));
91          return this;
92      }
93  
94      Stream<DynamicTest> generateTests(String template, Assertion8<A, B, C, D, E, F, G, H> verification) {
95          return data.stream().map(row -> {
96              String title = template.replace("#" + headers.getA(), String.valueOf(row.getA()));
97              title = title.replace("#" + headers.getB(), String.valueOf(row.getB()));
98              title = title.replace("#" + headers.getC(), String.valueOf(row.getC()));
99              title = title.replace("#" + headers.getD(), String.valueOf(row.getD()));
100             title = title.replace("#" + headers.getE(), String.valueOf(row.getE()));
101             title = title.replace("#" + headers.getF(), String.valueOf(row.getF()));
102             title = title.replace("#" + headers.getG(), String.valueOf(row.getG()));
103             String finalTitle = title.replace("#" + headers.getH(), String.valueOf(row.getH()));
104             return DynamicTest.dynamicTest(
105                     finalTitle,
106                     () -> {
107                         if (!verification.verify(row.getA(), row.getB(), row.getC(), row.getD(), row.getE(), row.getF(), row.getG(), row.getH())) {
108                             throw new AssertionFailedError("Verification failed for " + finalTitle + " with values " + headers.getA() + "=" + row.getA() + ", " + headers.getB() + "=" + row.getB() + ", " + headers.getC() + "=" + row.getC() + ", " + headers.getD() + "=" + row.getD() + ", " + headers.getE() + "=" + row.getE() + ", " + headers.getF() + "=" + row.getF() + ", " + headers.getG() + "=" + row.getG() + ", " + headers.getH() + "=" + row.getH());
109                         }
110                     });
111         });
112     }
113 
114 }
